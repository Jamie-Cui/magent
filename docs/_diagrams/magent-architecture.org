* One Magent Turn

#+begin_src plantuml :file ../assets/img/diagrams/one-magent-turn.svg :exports results :results file
@startuml
title One Magent turn
skinparam monochrome true
skinparam shadowing false

actor User
participant "agent-shell" as UI
participant "magent-runtime-api.el" as Runtime
participant "magent-runtime-queue.el" as Queue
participant "magent-ledger.el" as Ledger
participant "magent-agent.el" as Agent
participant "magent-llm-gptel.el" as Adapter
participant "gptel-request" as Gptel
participant "magent-agent-loop.el" as Loop
participant "magent-tool-orchestrator.el" as Orchestrator
participant "magent-tools.el" as Tools

User -> UI : prompt
UI -> Runtime : submit(prompt, session, observer)
Runtime -> Ledger : queue turn + user item
Runtime -> Queue : enqueue submission
Queue -> Runtime : start when active slot free
Runtime -> Agent : run turn
Agent -> Ledger : build prompt from completed turns + current turn
Agent -> Adapter : sample normalized request
Adapter -> Gptel : gptel-request
Gptel --> Adapter : text / reasoning / tool-use / done
Adapter --> Loop : normalized events

alt tool-call-batch-end
  Loop -> Orchestrator : dispatch tool calls
  Orchestrator -> Orchestrator : resolve permission / approval / audit
  Orchestrator -> Tools : execute serially
  Tools --> Orchestrator : result
  Orchestrator --> Loop : tool output
  Loop -> Ledger : update tool item
  Loop -> Agent : needs follow-up sampling
  Agent -> Ledger : rebuild prompt with tool output
  Agent -> Adapter : continue sampling
else completed
  Loop -> Ledger : complete assistant item + turn
  Loop --> Runtime : final result
  Runtime --> UI : session/update
end
@enduml
#+end_src
